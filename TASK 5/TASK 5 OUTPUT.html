import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load dataset (check column names before specifying usecols)
file_path = r"D:\PRODIGY\TASK 5\US_Accidents_March23.csv"

df = pd.read_csv(file_path, nrows=100000)  # Load first 100K rows


print("Columns in dataset:", df.columns)


if 'Start_Time' in df.columns:
    df['Start_Time'] = pd.to_datetime(df['Start_Time'], errors='coerce')
    df.dropna(subset=['Start_Time'], inplace=True)
    df['Hour'] = df['Start_Time'].dt.hour
    df['Weekday'] = df['Start_Time'].dt.day_name()
else:
    print("Error: 'Start_Time' column is missing!")
    exit()


plt.figure(figsize=(10, 5))
sns.countplot(x='Hour', data=df, palette='coolwarm')
plt.title("Accident Frequency by Hour")
plt.xlabel("Hour of the Day")
plt.ylabel("Accident Count")
plt.xticks(rotation=45)
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.show()


plt.figure(figsize=(10, 5))
sns.countplot(x='Weekday', data=df, order=['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday', 'Sunday'], palette='magma')
plt.title("Accident Frequency by Weekday")
plt.xlabel("Day of the Week")
plt.ylabel("Accident Count")
plt.xticks(rotation=45)
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.show()


hour_weekday_pivot = df.pivot_table(index='Weekday', columns='Hour', values='Start_Time', aggfunc='count', fill_value=0)

plt.figure(figsize=(10, 5))
sns.heatmap(hour_weekday_pivot, cmap="YlGnBu", linewidths=0.5, annot=False)
plt.title("Heatmap of Accidents by Hour and Weekday")
plt.xlabel("Hour of the Day")
plt.ylabel("Day of the Week")
plt.show()
